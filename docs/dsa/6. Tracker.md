20. https://leetcode.com/problems/find-numbers-with-even-number-of-digits/
```cpp
class Solution {
public:
    int findNumbers(vector<int>& nums) {
        int cnt = 0;
        for(auto i: nums){
            int d = log10(i) + 1;
            if(d % 2 == 0){
                cnt++;
            }
        }
        return cnt;
    }
};
```
21. https://leetcode.com/problems/gcd-of-odd-and-even-sums/
      - Sum of first `n`th even number = n(n+1)
      - Sum of first `n`th odd number = n*n
```cpp
class Solution {
public:
    int gcdOfOddEvenSums(int n) {
        return __gcd(n* (n + 1), (n* n));
    }
};
```
